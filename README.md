<h1>Install Project</h1>
<hr>
git clone<br> 
php init<br>
composer install<br>
php artisan migrate<br>
<i>development окружение</i><br>
флаг APP_ENV=local<br> 
флаг APP_DEBUG=true<br>
<i>production окружение</i><br>
флаг APP_ENV=production <br> 
флаг APP_DEBUG=false<br>

<h1>npm</h1>
<hr>
$npm install<br>
<i>development окружение</i><br>
$npm run dev <br>
<i>production окружение</i><br>
$npm run prod
